% tint range in the PDF
\newcommand{\CellTintMin}{5}
\newcommand{\CellTintMax}{45}

% data range for scaling
\newcommand{\CellDataMax}{100}

\newcommand{\setcellcolorrange}[3]{%
    \renewcommand{\CellTintMin}{#1}%
    \renewcommand{\CellTintMax}{#2}%
    \renewcommand{\CellDataMax}{#3}%
}

\newcommand{\autocolorcell}[2]{%
    \pgfmathparse{round(min(\CellTintMax, max(\CellTintMin, \CellTintMin + abs(#1)/\CellDataMax * (\CellTintMax-\CellTintMin))))}%
    \edef\shade{\pgfmathresult}%
    \ifdim #1pt>0pt
        \edef\tempcolor{green!\shade}%
    \else
        \edef\tempcolor{red!\shade}%
    \fi
    \expandafter\cellcolor\expandafter{\tempcolor}#2%
}
